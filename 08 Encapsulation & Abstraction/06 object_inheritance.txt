public class Library {

    // List to store Book objects
    private List<Book> books;

    // Constructor
    public Library() {
        books = new List<Book>();
    }

    // Add a book
    public void addBook(Book b) {
        books.add(b);
    }

    // Display all books
    public void displayBooks() {
        if (books.isEmpty()) {
            System.debug('The library is empty.');
            return;
        }
        
        for (Book b : books) {
            b.displayBookDetails(); // Call the method from Book class
        }
    }
}
